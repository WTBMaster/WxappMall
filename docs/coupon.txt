# 优惠券
------

coupon_spec {
    _id,
    enable      // true  or  false
    type        // 折扣 或者 满减
    value
    tag
    desc        // 满多少钱可用  无门槛使用
    title       // 什么的折扣券
    begin       // 开始时间
    deadline    // 到期时间
}
